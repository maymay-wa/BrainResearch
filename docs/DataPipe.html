 82        """
 83        Purpose:
 84        •	Adds new columns to the participant DataFrame:
 85        •	File paths for MRI images.
 86        •	Volume calculations per brain region.
 87        •	Changes in volume between baseline and follow-up scans.
 88

105            new_cols[f"{label} Change"] = [None]*len(self.participants_df)
106
107        # Create an empty DataFrame and horizontally concatenate
108        columns_df = pd.DataFrame(new_cols, index=self.participants_df.index)
109        self.participants_df = pd.concat([self.participants_df, columns_df], axis=1)

171    def register_and_convert_to_nifti(self, fixed_image, moving_image, subject_id, session, transform_type='Affine'):
172        """
173        Registers `moving_image` to `fixed_image` using ANTs, writes out the
174        transformed result to disk as a NIfTI file, and returns the nibabel image.
175
176        Purpose:
177        •	Uses ANTs to register a subject’s MRI scan to a reference (atlas).
178        •	Saves the registered image as a NIfTI file.
179        •	Handles errors and skips re-registration if output already exists.
180
181        Logic:
182        1.	Check if the output file exists.
183        2.	Apply bias field correction and smoothing.
184        3.	Perform image registration using ANTs.
185        4.	Save the registered image to disk.
186        5.	Return a nibabel-compatible image.
187        """
188        out_path = f"output/registered_output_sub-{subject_id}_ses-{session}.nii.gz"
189        
190        # If the file already exists, load it and skip registration
191        if os.path.exists(out_path):
192            try:
193                file = nib.load(out_path)
194                return file
195            except Exception as e:
196                logging.error(f"Error loading existing registered image for subject {subject_id}, session {session}: {e}")
197                raise
198        
199        # Preprocess the moving image (bias field correction + smoothing)
200        moving_image = ants.n4_bias_field_correction(moving_image)
201        moving_image = ants.smooth_image(moving_image, 2)
202
203        # Registration
204        try:
205            outputImage = ants.registration(
206                fixed=fixed_image,
207                moving=moving_image,
208                type_of_transform=transform_type
209            )['warpedmovout']
210        except Exception as e:
211            logging.error(f"Error registering images for subject {subject_id}, session {session}: {e}")
212            raise
213
214        # Write out the registered result
215        ants.image_write(outputImage, out_path)
216        
217        # Load the resulting file as a nibabel image
218        try:
219            file = nib.load(out_path)
220            return file
221        except Exception as e:
222            logging.error(f"Error loading registered image from disk for subject {subject_id}, session {session}: {e}")
223            raise

225    def loadImage(self, imgPath, subject_id, session):
226        """
227        Reads an image from a given path, then registers it against the
228        Harvard-Oxford atlas, returning a NIfTI image resampled to the atlas space.
229
230        Purpose:
231        •	Reads an MRI image from disk.
232        •	Registers it against the Harvard-Oxford atlas.
233        •	Ensures consistency in shape and resolution using resampling.
234
235        Key Steps:
236        1.	Load the Harvard-Oxford Atlas.
237        2.	Read the subject’s MRI image.
238        3.	Register the image to the atlas space.
239        4.	Resample to match the atlas resolution.
240        """
241        # Path to your local Harvard-Oxford atlas file if needed:
242        # (Should match what's inside your `fetch_atlas_harvard_oxford` location)
243        atlasPath = '/Users/mayerunterberg/nilearn_data/fsl/data/atlases/HarvardOxford/HarvardOxford-cort-maxprob-thr25-1mm.nii.gz'
244        atlas_image = ants.image_read(atlasPath)
245        ants_img = ants.image_read(imgPath)
246        
247        # Register and convert to NIfTI
248        nifti = self.register_and_convert_to_nifti(
249            fixed_image=atlas_image,
250            moving_image=ants_img,
251            subject_id=subject_id,
252            session=session
253        )
254
255        # Resample the result to the official atlas image just in case
256        # (the loaded self.ATLAS_IMAGE might be the same, but we do this 
257        #  to ensure consistent shape/resolution)
258        img_resampled = resample_to_img(
259            source_img=nifti,
260            target_img=self.ATLAS_IMAGE,
261            force_resample=True,
262            copy_header=True,
263            interpolation='nearest'
264        )
265        return img_resampled

267    def findDifferingAreasAndVolume(self, index, img1, img2, threshold=None):
268        """
269        Given two images, computes the difference map, calculates mean difference
270        and volume changes per region, and records those metrics in the DataFrame.
271
272        •	Computes difference maps between baseline and follow-up MRI images.
273        •	Calculates mean intensity differences in brain regions.
274        •	Estimates volume changes per region.
275        •	Stores results in self.participants_df.
276
277        Key Steps:
278        1.	Create a difference map: diff = math_img("img1 - img2", img1=img1, img2=img2)
279        2.	Extract region-wise voxel intensities.
280        3.	Compute mean intensity change per region.
281        4.	Calculate volume differences based on voxel count.
282        5.	Update the participant DataFrame.
283
284        """
285        diff = math_img("img1 - img2", img1=img1, img2=img2)
286        diff_data = get_data(diff)
287        
288        # For volume calculations, get data for each image
289        img1_data = get_data(img1)
290        img2_data = get_data(img2)
291        
292        # Use the ATLAS_IMAGE affine to find voxel volume
293        voxel_sizes = np.abs(np.diag(self.ATLAS_IMAGE.affine)[:3])  # shape (3,)
294        voxel_volume = np.prod(voxel_sizes)  # in mm^3
295
296        for label in self.REGION_LABELS:
297            # region_name from self.LABEL_NAMES
298            if 0 <= label < len(self.LABEL_NAMES):
299                region_name = self.LABEL_NAMES[label]
300            else:
301                region_name = "Unknown Region"
302                continue
303            
304            if region_name in ("Background", "Unknown Region"):
305                continue
306
307            regionMask = (self.ATLAS_DATA == label)
308            
309            # Mean difference in region
310            regionChanges = diff_data[regionMask]
311            region_mean_diff = round(float(np.mean(regionChanges)), 2)
312            
313            # Volume difference (count of non-zero or above-threshold voxels)
314            if threshold is not None:
315                img1_masked = ((img1_data > threshold) & regionMask)
316                img2_masked = ((img2_data > threshold) & regionMask)
317            else:
318                img1_masked = ((img1_data != 0) & regionMask)
319                img2_masked = ((img2_data != 0) & regionMask)
320            
321            # Filters our zero voxels
322            region_img1_voxels = np.count_nonzero(img1_masked)
323            region_img2_voxels = np.count_nonzero(img2_masked)
324            
325            # Calculates region volume
326            region_img1_volume = region_img1_voxels * voxel_volume
327            region_img2_volume = region_img2_voxels * voxel_volume
328            region_volume_diff = round(region_img1_volume - region_img2_volume, 2)
329            avgVolume = round((region_img1_volume + region_img2_volume) / 2, 2)
330            
331            # Save to the DataFrame
332            self.participants_df.loc[index, f"{region_name} Change"] = region_mean_diff
333            self.participants_df.loc[index, f"{region_name} Volume Avg"] = avgVolume
334            self.participants_df.loc[index, f"{region_name} Volume Change"] = region_volume_diff

371    def get_roi_differences_for_subject(self, subject_id):
372        """
373        Returns a dictionary of {RegionName: difference_value} for the given subject.
374        Assumes 'self.participants_df' contains columns like 'Frontal Pole Change', etc.
375        """
376
377        # 1. Locate the row for this subject
378        row_mask = self.participants_df['participant_id'] == subject_id
379        if not any(row_mask):
380            print(f"No data found for subject {subject_id}")
381            return {}
382
383        # Extract the row for the subject
384        subject_row = self.participants_df.loc[row_mask].iloc[0]
385
386        # 2. Gather all columns that end with "Change"
387        #    (These columns were created by findDifferingAreasAndVolume)
388        change_columns = ['Frontal Pole Change', 'Parahippocampal Gyrus, posterior division Change', 'Middle Temporal Gyrus, posterior division Volume Change', 'Inferior Temporal Gyrus, temporooccipital part Change']
389
390        # 3. Build a dictionary: region_name -> difference
391        #    Each column has the format "Frontal Pole Change", "Occipital Lobe Change", etc.
392        roi_differences = {}
393        for col in change_columns:
394            # Region name is the column name minus the suffix " Change"
395            region_name = col.replace(" Change", "")
396            roi_differences[region_name] = subject_row[col]
397
398        return roi_differences

268    def findDifferingAreasAndVolume(self, index, img1, img2, threshold=None):
269        """
270        Given two images, computes the difference map, calculates mean difference
271        and volume changes per region, and records those metrics in the DataFrame.
272
273        •	Computes difference maps between baseline and follow-up MRI images.
274        •	Calculates mean intensity differences in brain regions.
275        •	Estimates volume changes per region.
276        •	Stores results in self.participants_df.
277
278        Key Steps:
279        1.	Create a difference map: diff = math_img("img1 - img2", img1=img1, img2=img2)
280        2.	Extract region-wise voxel intensities.
281        3.	Compute mean intensity change per region.
282        4.	Calculate volume differences based on voxel count.
283        5.	Update the participant DataFrame.
284
285        """
286        diff = math_img("img1 - img2", img1=img1, img2=img2)
287        diff_data = get_data(diff)
288        
289        # For volume calculations, get data for each image
290        img1_data = get_data(img1)
291        img2_data = get_data(img2)
292        
293        # Use the ATLAS_IMAGE affine to find voxel volume
294        voxel_sizes = np.abs(np.diag(self.ATLAS_IMAGE.affine)[:3])  # shape (3,)
295        voxel_volume = np.prod(voxel_sizes)  # in mm^3
296
297        for label in self.REGION_LABELS:
298            # region_name from self.LABEL_NAMES
299            if 0 <= label < len(self.LABEL_NAMES):
300                region_name = self.LABEL_NAMES[label]
301            else:
302                region_name = "Unknown Region"
303                continue
304            
305            if region_name in ("Background", "Unknown Region"):
306                continue
307
308            regionMask = (self.ATLAS_DATA == label)
309            
310            # Mean difference in region
311            regionChanges = diff_data[regionMask]
312            region_mean_diff = round(float(np.mean(regionChanges)), 2)
313            
314            # Volume difference (count of non-zero or above-threshold voxels)
315            if threshold is not None:
316                img1_masked = ((img1_data > threshold) & regionMask)
317                img2_masked = ((img2_data > threshold) & regionMask)
318            else:
319                img1_masked = ((img1_data != 0) & regionMask)
320                img2_masked = ((img2_data != 0) & regionMask)
321            
322            # Filters our zero voxels
323            region_img1_voxels = np.count_nonzero(img1_masked)
324            region_img2_voxels = np.count_nonzero(img2_masked)
325            
326            # Calculates region volume
327            region_img1_volume = region_img1_voxels * voxel_volume
328            region_img2_volume = region_img2_voxels * voxel_volume
329            region_volume_diff = round(region_img1_volume - region_img2_volume, 2)
330            avgVolume = round((region_img1_volume + region_img2_volume) / 2, 2)
331            
332            # Save to the DataFrame
333            self.participants_df.loc[index, f"{region_name} Change"] = region_mean_diff
334            self.participants_df.loc[index, f"{region_name} Volume Avg"] = avgVolume
335            self.participants_df.loc[index, f"{region_name} Volume Change"] = region_volume_diff
